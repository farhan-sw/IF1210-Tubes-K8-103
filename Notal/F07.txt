{ Fungsi mengenerate pencarian bahan bangunan dari 0-5 dan akan mengupdate data bahan bangunan }
function kumpul (input data_bahan : array of array of string, input NMax : int, input isPrint : bool = True) -> data_bahan : list[list[str]]
{ Input : data bahan }
{ Output : data bahan yang telah dimodifikasi penambahan jin }

KAMUS LOKAL 
    rand_pasir  : int
    rand_batu   : int
    rand_air    : int
    tmp         : int
    function randomNumber(start : integer, stop : integer) -> integer
    { Mengembalikan nilai acak diantara start - stop }

ALGORITMA
    rand_pasir  <- B01_RNG.randomNumber(0, 5)          { Ambil data random untuk pasir, batu, air dari 0-5 }
    rand_batu   <- B01_RNG.randomNumber(0, 5)
    rand_air    <- B01_RNG.randomNumber(0, 5)
    tmp         <- 0                                   { Variabel penampung sementara str-int }

    { Update Data Bahan untuk pasir, batu, air }

    tmp <- 0                                           { Reset variabel }
    for i <- 0 to NMax-1 do                             { Update pasir, cari lokasi pasir }
        if (data_bahan[i][0] = "pasir") then
            tmp <- int(data_bahan[i][2])                 { Konversi string ke integer }
            tmp <- tmp + rand_pasir
            data_bahan[i][2] <- str(tmp)                 { Update nilai }

    tmp <- 0                                           { Reset variabel }
    for i <- 0 to NMax-1 do                             { Update batu, cari lokasi batu }
        if (data_bahan[i][0] = "batu") then
            tmp <- int(data_bahan[i][2])                 { Konversi string ke integer }
            tmp <- tmp + rand_batu
            data_bahan[i][2] <- str(tmp)                 { Jumlah Batu berada di kolom indeks 2 }

    tmp <- 0                                           { Reset variabel }
    for i <- 0 to NMax-1 do                             { Update air, cari lokasi air }
        if (data_bahan[i][0] = "air") then
            tmp <- int(data_bahan[i][2])                 { Konversi string ke integer }
            tmp <- tmp + rand_air
            data_bahan[i][2] <- str(tmp)                 { Jumlah air berada di kolom indeks 2 }

    if (isPrint) then
        output ("Jin menemukan", rand_pasir, " pasir,", rand_batu, " batu, dan", rand_air, " air.")

    ->  data_bahan
