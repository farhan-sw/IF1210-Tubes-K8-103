function randomNumber(start : integer, stop : integer) -> integer
{ Fungsi menerima dua parameter start dan stop }
{ Fungsi mengembalikan angka acak antara start dan stop }

KAMUS LOKAL :
    seed    : integer
    m       : integer
    a       : integer
    c       : integer
    result  : integer

ALGORITMA
    sleep(0.005) { Memberi jeda waktu 0.005 detik }
    seed <- int(time() * 1000) { Membuat seed }
    m <- 2^32                                   { Parameter LCG }
    a <- 1103515245
    c <- 12345

    seed <- (a * seed + c) % m                   { Menghasilkan angka acak berdasarkan LCG }

    result <- start + (seed % (stop - start))    { Menghasilkan angka di antara start dan stop }

    -> result
